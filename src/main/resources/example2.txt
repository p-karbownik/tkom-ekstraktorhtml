resource
{
	tag = [img];

	conditions
	{
		if parent.parent has class == [media-content];
		if parent has attribute == [title];
		if this has attribute==[alt];
	}
	export to class = [Link];
	set fields
	{
		field[link_to_image] = from this.attribute[src].text;
		field[description] = from this.attribute[alt].text;
	}

	amount = every;
}
